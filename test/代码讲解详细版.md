## 一、项目入口与启动流程 (`main.py`)

### 1.1 完整启动流程

```python
# main.py — 完整代码讲解

# ========== 第一步：环境预配置 ==========
# 必须在导入 transformers 之前完成
# 原因：huggingface_hub 在导入时会缓存 HF_ENDPOINT 环境变量

_dotenv_path = os.path.join(_backend_dir, ".env")
if os.path.exists(_dotenv_path):
    from dotenv import load_dotenv
    load_dotenv(_dotenv_path)  # 加载 .env 文件中的环境变量

# 设置 HuggingFace 镜像（国内加速下载）
if "HF_ENDPOINT" not in os.environ:
    os.environ["HF_ENDPOINT"] = "https://hf-mirror.com"

# ========== 第二步：预加载深度学习框架 ==========
try:
    from transformers import AutoTokenizer, AutoModelForCausalLM
    import torch
    logging.getLogger(__name__).info("transformers/torch 预加载成功")
except ImportError as e:
    logging.getLogger(__name__).warning(f"transformers/torch 预加载失败: {e}")

# ========== 第三步：创建 FastAPI 应用 ==========
app = FastAPI(
    title="会议纪要智能体",
    description="智能会议纪要助手 - 自动转写、提取、生成结构化会议纪要",
    version="1.0.0",
)

# ========== 第四步：配置 CORS 跨域 ==========
app.add_middleware(
    CORSMiddleware,
    allow_origins=ALLOWED_ORIGINS,  # 允许的前端域名
    allow_credentials=True,         # 允许携带 Cookie
    allow_methods=["*"],            # 允许所有 HTTP 方法
    allow_headers=["*"],            # 允许所有请求头
)

# ========== 第五步：注册路由 ==========
app.include_router(meetings.router)    # 会议管理 API
app.include_router(minutes.router)     # 纪要管理 API
app.include_router(export.router)      # 导出 API
app.include_router(auth.router)        # 认证 API
app.include_router(templates.router)   # 模板 API
app.include_router(websocket.router)   # WebSocket API
app.include_router(audio.audio_router) # 音频处理 API
app.include_router(public.router)      # 公开文档 API
app.include_router(qa_router)          # 智能问答 API

# ========== 第六步：挂载静态文件 ==========
if os.path.exists(FRONTEND_DIR):
    app.mount("/static", StaticFiles(directory=FRONTEND_DIR), name="static")

# ========== 第七步：启动时初始化 ==========
@app.on_event("startup")

    # 创建数据库表
    Base.metadata.create_all(bind=engine)
    
    # 后台线程预加载 Whisper 模型（避免首次请求延迟）
    def _preload():
        from app.services.speech_service import _get_faster_whisper_model
        m = _get_faster_whisper_model()
        logger.info("Model pre-loaded: %s", "OK" if m else "FAILED")
    
    threading.Thread(target=_preload, daemon=True).start()

# ========== 第八步：首页路由 ==========
@app.get("/")
async def index():
    index_path = os.path.join(FRONTEND_DIR, "index.html")
    if os.path.exists(index_path):
        return FileResponse(index_path, headers={
            "Cache-Control": "no-cache, no-store, must-revalidate",
            "Pragma": "no-cache",
            "Expires": "0"
        })
    return {"message": "会议纪要智能体API服务运行中", "docs": "/docs"}
```

### 1.2 关键设计决策

| 决策 | 原因 |
|------|------|

```python
# database.py — 完整代码讲解

import time
import logging
from sqlalchemy import create_engine, event
from sqlalchemy.orm import sessionmaker, declarative_base
from config import DATABASE_URL

# ========== 创建数据库引擎 ==========
engine = create_engine(
    DATABASE_URL,
    connect_args={"check_same_thread": False},  # SQLite 允许多线程访问
    pool_pre_ping=True,   # 每次使用前检测连接是否有效
    pool_size=5,          # 连接池大小（保持 5 个空闲连接）
    max_overflow=5,       # 溢出连接数（最多 10 个并发连接）
    pool_recycle=3600,    # 连接回收时间（1 小时）
    echo=False,           # 不打印 SQL 语句（生产环境）
)

# ========== SQLite 性能优化 ==========
@event.listens_for(engine, "connect")
def _set_sqlite_pragma(dbapi_connection, connection_record):
    """
    SQLite 性能优化 PRAGMA 设置
    
    WAL (Write-Ahead Logging) 模式：
    - 允许读写并发（普通模式下写操作会阻塞读操作）
    - 提升并发性能 2-5 倍
    - 数据安全性更高（崩溃恢复更快）
    """
    cursor = dbapi_connection.cursor()
    cursor.execute("PRAGMA journal_mode=WAL")      # WAL 模式
    cursor.execute("PRAGMA synchronous=NORMAL")     # 平衡性能和安全
    cursor.execute("PRAGMA cache_size=-64000")      # 64MB 缓存（负数表示 KB）
    cursor.execute("PRAGMA temp_store=MEMORY")      # 临时表存储在内存
    cursor.close()

# ========== 慢查询日志 ==========
@event.listens_for(engine, "before_cursor_execute")
def _log_slow_queries(conn, cursor, statement, parameters, context, executemany):
    """记录查询开始时间"""
    conn.info.setdefault("query_start_time", []).append(time.monotonic())

@event.listens_for(engine, "after_cursor_execute")
def _log_query_duration(conn, cursor, statement, parameters, context, executemany):
    """记录查询耗时，超过 1 秒的记录为慢查询"""
    start_times = conn.info.get("query_start_time", [])
    if start_times:
        start_time = start_times.pop()
        duration = time.monotonic() - start_time
        if duration > 1.0:
            logger.warning("Slow query (%.2fs): %s", duration, statement[:200])

# ========== 创建会话工厂 ==========
SessionLocal = sessionmaker(autocommit=False, autoflush=False, bind=engine)
Base = declarative_base()

def get_db():
    """
    FastAPI 依赖注入：获取数据库会话
    
    使用方式：
    @router.get("/api/meetings")
    async def list_meetings(db: Session = Depends(get_db)):
        meetings = db.query(Meeting).all()
    """
    db = SessionLocal()
    try:
        yield db  # 返回会话给请求处理
    finally:
        db.close()  # 请求结束后关闭会话

def init_db():
    """初始化数据库表"""
    Base.metadata.create_all(bind=engine)
```

## 四、语音识别模块 (`speech_service.py`)

### 4.1 双引擎架构设计

```python
# speech_service.py — 核心架构

class SpeechRecognizer:
    """
    语音识别服务 - 双引擎架构
    
    支持两种 ASR 引擎：
    1. faster-whisper：本地推理，免费，隐私安全
    2. MiMo ASR：云端推理，准确率高，支持多方言
    
    选择策略：
    - 用户指定 whisper → 使用本地 whisper
    - 用户指定 mimo → 使用云端 MiMo
    - 自动选择 → 优先 whisper（免费），失败降级到 mimo
    """
    
    def recognize_file(self, filepath: str, asr_engine: str = "") -> dict:
        """
        语音识别主入口
        
        Args:
            filepath: 音频文件路径
            asr_engine: 引擎选择 ("whisper" / "mimo" / "" 自动)
        
        Returns:
            {"success": bool, "text": str, "engine": str}
        """
        logger.info(f"Speech recognition started | engine={asr_engine or 'auto'} | file={filepath}")
        
        # 第一步：音频预处理（可选增强）
        enhanced_path = self._enhance_if_needed(filepath)
        
        # 第二步：根据引擎选择识别方法
        result = None
        
        if asr_engine == "whisper" or not asr_engine:
            try:
                result = self._recognize_with_whisper(enhanced_path)
                result["engine"] = "whisper"
            except Exception as e:
                logger.warning(f"Whisper recognition failed: {e}")
                if asr_engine == "whisper":
                    return {"success": False, "message": f"Whisper 识别失败: {str(e)}"}
                # 自动模式下降级到 MiMo
                logger.info("Falling back to MiMo ASR")
        
        if result is None and (asr_engine == "mimo" or not asr_engine):
            try:
                result = self._recognize_with_mimo(enhanced_path)

            except Exception as e:

        # 第三步：后处理
        if result and result.get("text"):
            cleaned_text = self._post_process(result["text"])
            result["text"] = cleaned_text
        
        return result
```

```python
# ========== 幻觉检测 ==========
REPEAT_CHAR = re.compile(r'(.)\1{4,}')      # 单字符重复 5 次以上
REPEAT_WORD = re.compile(r'(\S{2,6})\1{3,}')  # 2-6 字符词重复 3 次以上
REPEAT_PHRASE = re.compile(r'(.{2,10})\1{2,}')  # 短语重复 2 次以上

WHISPER_INITIAL_PROMPT = "以下是一段中文会议录音的文字记录。"
WHISPER_PROMPT_FRAGMENTS = [
    "中文会议录音", "一段中文", "会议录音的文字记录",
    "中文会议", "以下是一段", "文字记录",
]

def detect_prompt_echo(text: str) -> bool:
    """
    检测 Whisper 是否只输出了初始提示词
    
    场景：音频内容为空或无法识别时，Whisper 可能只返回提示词
    """
    if not text:
        return True
    if text.strip().rstrip("。") == WHISPER_INITIAL_PROMPT.rstrip("。"):
        return True
    if len(text) < 40 and any(frag in text for frag in WHISPER_PROMPT_FRAGMENTS):
        return True
    return False

def detect_repetition(text: str) -> bool:
    """
    检测文本中的重复幻觉
    
    常见幻觉模式：
    - 单字符重复：哈哈哈哈哈哈哈
    - 词语重复：那个那个那个那个
    - 短语重复：对吧对吧对吧
    """
    if not text or len(text) < 4:
        return False
    if REPEAT_CHAR.search(text):
        return True
    if REPEAT_WORD.search(text):
        return True
    if REPEAT_PHRASE.search(text):
        return True
    return False

def _filter_whisper_hallucinations(self, text: str) -> str:
    """
    过滤 Whisper 常见幻觉输出
    
    过滤策略：
    1. Prompt Echo 检测：过滤初始提示词回显
    2. 重复检测：检测并移除重复文本
    3. 无意义内容过滤：过滤过短或无意义的句子
    """
    # 1. 检测 prompt echo
    if detect_prompt_echo(text):
        logger.warning("Prompt echo detected, returning empty")
        return ""
    
    # 2. 检测重复文本
    if detect_repetition(text):
        logger.warning("Repetition detected, cleaning text")
        text = self._remove_repetitions(text)
    
    # 3. 过滤无意义短句
    lines = text.split('\n')
    filtered = []
    for line in lines:
        line = line.strip()
        # 过滤过短的句子（少于 3 个字符）
        if len(line) < 3:
            continue
        # 过滤纯数字行
        if re.match(r'^[\d:：\s]+$', line):
            continue
        # 过滤纯标点行
        if re.match(r'^[，,。.！!？?]+$', line):
            continue
        filtered.append(line)
    
    return '\n'.join(filtered)
```

```python
# nlp_service.py — 核心架构

class NLPProcessor:
    """
    NLP 信息提取 - 规则匹配 + AI 推理混合架构
    
    设计理念：
    - 规则匹配：快速、确定性强、无需调用 API
    - AI 推理：理解能力强、处理复杂语义
    
    两者互补：
    - 规则匹配处理 80% 的简单场景
    - AI 推理处理 20% 的复杂场景
    """
    
    def process(self, text: str, title: str = "", participants: str = "", 
                max_tokens: int = 0, template_structure: str = None) -> dict:
        """
        主处理流程
        
        Args:
            text: 会议原始文本
            title: 会议标题（可选）
            participants: 参会人员（可选）
            max_tokens: 最大 token 数（影响 AI 推理深度）
            template_structure: 模板结构（可选）
        
        Returns:
            {
                "topic": "会议议题",
                "participants": "参会人员",
                "overview": "会议概述",
                "discussion_points": "讨论要点",
                "key_decisions": "关键决策",
                "action_items": [{"content": "...", "responsible_person": "...", "deadline": "..."}],
                "unresolved_issues": "遗留问题",
                "summary": "会议总结"
            }
        """
        # 1. 文本预处理
        clean_text = self.preprocess(text)
        
        # 2. 按发言人分组
        segments = self._group_segments_by_speaker(clean_text)
        
        # 3. 逐段提取信息
        all_decisions = []
        all_action_items = []
        all_issues = []
        all_points = []
        
        for seg in segments:
            # 口语连接词替换为句号
            block = _RE_MULTI_COMMA.sub('。', seg)
            block = re.sub(rf'(?:{_ORAL_PATTERN})', '。', block)
            
            # 按标点分句
            block_sents = [s.strip() for s in re.split(r'[。！？；;\n]+', block) if len(s.strip()) >= 3]
            
            for sent in block_sents:
                # 提取遗留问题
                is_issue = self._sentence_has_keywords(sent, _ISSUE_KW_RE)
                if is_issue:
                    if sent not in all_issues:
                        all_issues.append(sent)
                    continue
                
                # 提取关键决策
                if self._sentence_has_keywords(sent, _DECISION_KW_RE):
                    if sent not in all_decisions:
                        all_decisions.append(sent)
                
                # 提取行动项
                if self._sentence_has_keywords(sent, _ACTION_KW_RE):
                    item = self._extract_action_item(sent, block)
                    if item:
                        all_action_items.append(item)
            
            # 使用 jieba 提取关键词作为讨论要点
            if self.jieba_loaded:
                import jieba.analyse
                tags = jieba.analyse.extract_tags(block, topK=3)
                if tags:
                    all_points.append("、".join(tags))
        
        # 4. 去重
        seen_actions = set()
        unique_actions = []
        for item in all_action_items:
            key = (item["responsible_person"], item["content"][:30])
            if key not in seen_actions:
                seen_actions.add(key)
                unique_actions.append(item)
        
        # 5. 格式化输出
        decisions_str = "\n".join([f"• {d}" for d in all_decisions]) if all_decisions else "会议中未明确记录关键决策"
        issues_str = "\n".join([f"• {i}" for i in all_issues]) if all_issues else "无明确遗留问题记录"
        points_str = "\n".join([f"• {p}" for p in all_points[:8]]) if all_points else "详见会议原始记录"
        
        # 6. 提取参会人员
        ai_participants = self.extract_participants(clean_text, _corrected=True)
        final_participants = ai_participants or participants
        
        # 7. 生成总结
        summary = self.generate_summary(clean_text, decisions_str, unique_actions, issues_str)
        
        return {
            "topic": self.extract_topic(clean_text, title, _corrected=True),
            "participants": final_participants,
            "overview": self.extract_overview(clean_text, _corrected=True),
            "discussion_points": points_str,
            "key_decisions": decisions_str,
            "action_items": unique_actions,
            "unresolved_issues": issues_str,
            "summary": summary,
        }
```

```python
# ========== 关键词库 ==========
DECISION_KEYWORDS = [

]

    "问题", "未解决", "待讨论", "需要进一步", "遗留", "暂缓",
    "延后", "搁置", "尚需", "还需", "继续讨论", "下次会议"
]

# ========== 行动项提取 ==========
def _extract_action_item(self, sent: str, context: str) -> dict:
    """
    从句子中提取行动项
    
    提取内容：
    1. 责任人（通过正则匹配）
    2. 截止时间（通过正则匹配）
    3. 任务内容（清理后的句子）
    """
    # 1. 提取责任人
    person = ""
    
    # 人名匹配模式
    PERSON_PATTERNS = [
        r'(?:由|请|让|安排|指派|交给)([\u4e00-\u9fa5]{2,4}?)(?:负责|完成|跟进)',
        r'([\u4e00-\u9fa5]{2,4})(?:负责|来负责|来完成)',
        r'@([\u4e00-\u9fa5]{2,4})',
    ]
    
    for pattern in PERSON_PATTERNS:
        match = re.search(pattern, sent)
        if match:
            candidate = match.group(1)
            # 验证是否是有效人名
            if self._is_valid_person(candidate):
                person = candidate
                break
    
    # 如果正则匹配失败，使用 jieba 词性标注
    if not person:
        pos_persons = self._extract_persons_by_pos(sent)
        if pos_persons:
            person = pos_persons[0]
    
    # 2. 提取截止时间
    deadline = ""
    
    DEADLINE_PATTERNS = [
        r'(截止|deadline)[：:]\s*(.+?)(?:[。；\n]|$)',
        r'(\d+月\d+[日号])',
        r'(\d{4}[-/]\d{1,2}[-/]\d{1,2})',
        r'(下周[一二三四五六日])',
        r'(今天|明天|后天)',
    ]
    
    for pattern in DEADLINE_PATTERNS:
        match = re.search(pattern, sent)
        if match:
            deadline = match.group(0)
            # 清理前缀
            deadline = re.sub(r'^(截止|截止时间|截止日期|deadline)[：:]\s*', '', deadline)
            break
    
    # 3. 清理任务内容
    cleaned = re.sub(r'^[，,。.！!？?\s]+', '', sent)
    cleaned = re.sub(r'[，,。.！!？?\s]+$', '', cleaned)
    
    # 过滤过短或无意义的内容
    if not cleaned or len(cleaned) < 8:
        return None
    
    skip_words = {"好", "行", "对", "嗯", "啊", "可以", "没问题", "好的", "明白"}
    if cleaned in skip_words:
        return None
    
    return {
        "content": cleaned,
        "responsible_person": person,
        "deadline": deadline,
        "priority": "medium"
    }

    """
    使用 jieba 词性标注提取人名
    
    jieba 词性标注：
    - nr: 人名
    - ns: 地名
    - nt: 机构名
    """
    if not self.jieba_loaded:
        return []
    
    persons = []
    _PRONOUN_CHARS = set("我你他她它")
    
    for word, flag in pseg.cut(sent):
        if flag == "nr" and 2 <= len(word) <= 4:
            # 排除代词和常见非人名词汇
            if word in _NOT_PERSON_WORDS or word in _FILLER_WORDS:
                continue
            if any(ch in _PRONOUN_CHARS for ch in word):
                continue
            if _RE_PERSON_CHECK.match(word):
                persons.append(word)
    
    return persons
```

### 5.3 参会人员提取

```python
def extract_participants(self, text: str, _corrected: bool = False) -> str:
    """
    从文本中智能提取参会人员
    
    提取策略（按优先级）：
    1. 显式标签提取：如 "参会人：张三、李四"
    2. 说话人检测：如 "张三说..."
    3. 词性标注提取：jieba 词性标注识别的人名
    """
    if not _corrected:
        text = self._correct_homophone(text)
    
    persons = set()
    _PRONOUN_CHARS = set("我你他她它")
    
    # 策略 1：显式标签提取
    # 匹配 "参会人：张三、李四" 格式
    m = _RE_PARTICIPANT_LABEL.search(text)
    if m:
        for name in _RE_PARTICIPANT_SPLIT.split(m.group(1)):
            name = name.strip()
            if 2 <= len(name) <= 4 and _RE_PERSON_CHECK.match(name):
                if name not in _NOT_PERSON_WORDS:
                    persons.add(name)
    
    # 策略 2：说话人检测
    # 检测 "张三说"、"李四认为" 等模式
    for name in self._detect_speakers(text):
        if name not in _NOT_PERSON_WORDS and not any(ch in _PRONOUN_CHARS for ch in name):
            persons.add(name)
    
    # 策略 3：词性标注提取
    if self.jieba_loaded:
        for word, flag in pseg.cut(text):
            if flag == "nr" and 2 <= len(word) <= 4:
                if word not in _SKIP_WORDS and word not in _NOT_PERSON_WORDS:
                    if not any(ch in _PRONOUN_CHARS for ch in word):
                        persons.add(word)
    
    return "、".join(persons) if persons else ""

def _detect_speakers(self, text: str) -> list:
    """
    检测文本中的说话人
    
    检测模式：
    - "张三说..."
    - "李四认为..."
    - "王五：..."
    """
    if not self.jieba_loaded:
        return []
    
    speakers = []
    seen = set()
    _PRONOUN_CHARS = set("我你他她它")
    _VERB_CHARS = set("说讲问答来去做好行对开看搞定安排负责跟进")
    
    # 使用 jieba 词性标注
    for word, flag in pseg.cut(text):
        if flag == "nr" and 2 <= len(word) <= 4:
            if word in _NOT_PERSON_WORDS or word in _FILLER_WORDS or word in _SKIP_WORDS:
                continue
            if any(ch in _PRONOUN_CHARS for ch in word):
                continue
            if _RE_PERSON_CHECK.match(word) and word not in seen:
                seen.add(word)
                speakers.append(word)
    
    # 位置启发式：检查人名是否在句首或冒号前
    pos_set = set(speakers)
    for m in re.finditer(r'([\u4e00-\u9fa5]{2,3})', text):
        c = m.group(1)
        if c in _NOT_PERSON_WORDS or c in _SKIP_WORDS:
            continue
        if not _RE_PERSON_CHECK.match(c):
            continue
        if c in seen:
            continue
        
        # 检查位置
        end = m.end()
        before = text[m.start()-1:m.start()] if m.start() > 0 else ""
        after = text[end:end+1]
        at_boundary = (
            (not before or before in "。！？：:\n ") or
            (after in "，,。！？：:\n " or not after)
        )
        
        if at_boundary and (after in _VERB_CHARS or c in pos_set):
            seen.add(c)
            speakers.append(c)
    
    return speakers
```

        template_hint = ""
        if template_structure:
            template_hint = f"\n\n请按照以下会议结构模板提取信息：\n{template_structure}\n"

        if result is None:
            # 降级到关键词匹配
            logger.warning("DeepSeek API 调用失败，降级到关键词匹配")
            return _get_nlp_fallback().process(text, title, participants, template_structure=template_structure)

        return {
            "topic": result.get("topic", "会议记录"),
            "participants": result.get("participants", ""),
            "overview": result.get("overview", ""),
            "discussion_points": result.get("discussion_points", ""),
            "key_decisions": result.get("key_decisions", "会议中未明确记录关键决策"),

            "unresolved_issues": result.get("unresolved_issues", "无明确遗留问题记录"),
            "summary": result.get("summary", ""),
        }

            from openai import OpenAI

            client = OpenAI(
                api_key=self.api_key,
                base_url=self.base_url,
            )

            response = client.chat.completions.create(
                model=self.model,
                messages=[
                    {"role": "system", "content": DEEPSEEK_SYSTEM_PROMPT},
                    {"role": "user", "content": user_message},
                ],
                temperature=0.3,  # 低温度，输出更确定
                max_tokens=4096,
            )

            content = response.choices[0].message.content.strip()

            # 清理 markdown 代码块包裹

                lines = content.split("\n")
                if lines[0].startswith("```"):

class LocalQwenProcessor:
    """
    本地 Qwen3 模型处理器
    
    优势：
    - 无需网络
    - 隐私安全
    - 无调用费用
    
    劣势：
    - 需要下载模型（~1-2GB）
    - 推理速度较慢（CPU 约 30-60 秒）
    - 准确率可能低于云端模型
    """
    
    _model = None
    _tokenizer = None
    
    def _ensure_model(self):
        """
        懒加载模型
        
        首次调用时加载模型，后续复用缓存
        """
        if LocalQwenProcessor._model is not None:
            return  # 模型已加载

        import time as _time
        
        logger.info("Loading Qwen3 model...")
        t0 = _time.time()

        try:
            from transformers import AutoTokenizer, AutoModelForCausalLM
            import torch

            # 设置 CPU 线程数
            import multiprocessing
            cpu_count = multiprocessing.cpu_count()
            optimal_threads = min(cpu_count, 8)
            torch.set_num_threads(optimal_threads)

            # 加载 Tokenizer
            tokenizer = AutoTokenizer.from_pretrained(
                self.model_name,
                cache_dir=self.cache_dir,
                trust_remote_code=True,
            )

            # 根据设备选择加载方式
            if self.device == "cuda":
                # GPU 模式：尝试 4-bit 量化
                try:
                    import bitsandbytes
                    from transformers import BitsAndBytesConfig
                    
                    bnb_config = BitsAndBytesConfig(
                        load_in_4bit=True,
                        bnb_4bit_compute_dtype=torch.float16,
                        bnb_4bit_use_double_quant=True,
                    )
                    model = AutoModelForCausalLM.from_pretrained(
                        self.model_name,
                        cache_dir=self.cache_dir,
                        trust_remote_code=True,
                        device_map="auto",
                        quantization_config=bnb_config,
                        low_cpu_mem_usage=True,
                    )
                except Exception:
                    # 量化不可用，使用 float16
                    model = AutoModelForCausalLM.from_pretrained(
                        self.model_name,
                        cache_dir=self.cache_dir,
                        trust_remote_code=True,
                        torch_dtype=torch.float16,
                        device_map="auto",
                        low_cpu_mem_usage=True,
                    )
            else:
                # CPU 模式：使用 float32
                model = AutoModelForCausalLM.from_pretrained(
                    self.model_name,
                    cache_dir=self.cache_dir,
                    trust_remote_code=True,
                    torch_dtype=torch.float32,
                    low_cpu_mem_usage=True,
                )

            model.eval()  # 设置为评估模式

            # 缓存模型
            LocalQwenProcessor._tokenizer = tokenizer
            LocalQwenProcessor._model = model

            load_secs = _time.time() - t0
            logger.info(f"Qwen3 model loaded in {load_secs:.1f}s")

        except Exception as e:
            logger.error(f"Failed to load Qwen3 model: {e}")
            raise

    def _call_qwen(self, user_message: str, title: str = "", participants: str = "", 
                   max_tokens: int = 0, progress_callback=None) -> Optional[dict]:
        """
        使用 Qwen3 模型推理
        
        使用 TextIteratorStreamer 实现流式输出
        """
        import time as _time
        import threading
        import torch
        from transformers import TextIteratorStreamer

        self._log_request(user_message, title, participants)

        tokenizer = LocalQwenProcessor._tokenizer
        model = LocalQwenProcessor._model

        # 根据 max_tokens 选择 prompt 模式
        actual_max = max_tokens if max_tokens > 0 else LOCAL_MODEL_MAX_TOKENS
        if actual_max <= 300:
            system_prompt = QWEN_SYSTEM_PROMPT_BRIEF  # 简洁模式
        elif actual_max <= 800:
            system_prompt = QWEN_SYSTEM_PROMPT_NORMAL  # 标准模式
        else:
            system_prompt = QWEN_SYSTEM_PROMPT_DETAILED  # 详细模式

        messages = [
            {"role": "system", "content": system_prompt},
            {"role": "user", "content": user_message},
        ]
        
        # 使用 Qwen3 的 chat template
        prompt = tokenizer.apply_chat_template(
            messages,
            tokenize=False,
            add_generation_prompt=True,
            enable_thinking=False,  # 禁用思维链，加速推理
        )

        # Tokenize
        inputs = tokenizer(prompt, return_tensors="pt", truncation=True, max_length=32768)

        # 创建流式输出器
        streamer = TextIteratorStreamer(
            tokenizer, 
            skip_prompt=True,      # 跳过 prompt
            skip_special_tokens=True  # 跳过特殊 token
        )

        # 推理参数
        gen_kwargs = dict(
            **inputs,
            max_new_tokens=actual_max,
            do_sample=False,       # 贪心解码
            pad_token_id=tokenizer.eos_token_id,
            streamer=streamer,
        )

        # 在后台线程运行推理
        def _generate():
            with torch.inference_mode():  # 推理模式，节省内存
                model.generate(**gen_kwargs)

        gen_thread = threading.Thread(target=_generate, daemon=True)
        gen_thread.start()

        # 读取流式输出
        generated_text = ""
        token_count = 0

        for new_text in streamer:
            generated_text += new_text
            token_count += 1
            
            # 回调进度
            if progress_callback and token_count % 2 == 0:
                progress = min(token_count / actual_max, 0.99)
                progress_callback(progress, token_count, actual_max, generated_text)

        gen_thread.join()

        content = generated_text.strip()

        # 去除 <think>...</think> 思考过程（Qwen3 thinking 模型）
        if "<think>" in content and "</think>" in content:
            think_end = content.rfind("</think>")
            content = content[think_end + len("</think>"):].strip()

        # 去除 Markdown 代码块包裹
        if content.startswith("```"):
            lines = content.split("\n")
            if lines[0].startswith("```"):
                lines = lines[1:]
            if lines and lines[-1].startswith("```"):
                lines = lines[:-1]
            content = "\n".join(lines)

        # JSON 解析
        try:
            result = json.loads(content)
            return result
        except json.JSONDecodeError:
            logger.error(f"JSON 解析失败")
            return None

```python
class QAService:
    """
    基于 RAG 的会议智能问答服务
    
    RAG (Retrieval-Augmented Generation) 流程：
    1. 检索 (Retrieval)：从会议文本中检索相关段落
    2. 增强 (Augmentation)：将相关段落作为上下文
    3. 生成 (Generation)：基于上下文生成回答
    """
    
    _embedder = None  # 语义编码模型（类变量，全局共享）

    def __init__(self):
        self._ensure_embedder()

    def _ensure_embedder(self):

        if QAService._embedder is not None:
            return
        
        if not SENTENCE_TRANSFORMERS_AVAILABLE:
            logger.warning("sentence-transformers 未安装，语义检索不可用")
            return
        
        try:
            # paraphrase-multilingual-MiniLM-L12-v2：
            # - 支持 50+ 语言
            # - 384 维向量
            # - 适合语义相似度计算
            QAService._embedder = SentenceTransformer(
                "paraphrase-multilingual-MiniLM-L12-v2"
            )
            logger.info("语义检索模型加载完成")
        except Exception as e:
            logger.warning("语义检索模型加载失败: %s", e)

    def ask(self, meeting_id: int, question: str, db) -> dict:
        """
        问答主流程
        
        Args:
            meeting_id: 会议 ID
            question: 用户问题
            db: 数据库会话
        
        Returns:
            {
                "success": bool,
                "answer": str,
                "sources": list  # 来源段落
            }
        """
        if not question or not question.strip():
            return {"success": False, "message": "问题不能为空"}

        # 1. 获取会议原文
        meeting = db.query(Meeting).filter(Meeting.id == meeting_id).first()
        if not meeting:
            return {"success": False, "message": f"会议 ID {meeting_id} 不存在"}

        raw_text = meeting.raw_text or ""
        if not raw_text.strip():
            return {"success": False, "message": "该会议没有原始文本，无法回答问题"}

        # 2. 分割段落
        paragraphs = self._split_paragraphs(raw_text)
        
        # 3. 语义检索 Top-3 相关段落
        relevant = self._semantic_search(paragraphs, question, top_k=3)
        context = "\n".join(relevant)
        
        # 4. 生成回答
        answer = self._generate_answer(question, context)
        
        return {
            "success": True,
            "answer": answer,
            "sources": relevant,
            "meeting_id": meeting_id,
            "question": question,
        }
```

### 7.2 语义检索

```python
def _split_paragraphs(self, text: str, min_len: int = 10) -> list:
    """
    将文本分割为段落
    
    策略：
    1. 先按换行符分割
    2. 如果段落太少，按句子分割
    """
    paragraphs = []
    
    # 先按换行符分割
    for p in text.split("\n"):
        p = p.strip()
        if len(p) >= min_len:
            paragraphs.append(p)
    
    # 如果段落太少，按句子分割
    if not paragraphs:
        for sent in text.replace("。", "。\n").replace("！", "！\n").replace("？", "？\n").split("\n"):
            sent = sent.strip()
            if len(sent) >= min_len:
                paragraphs.append(sent)
    
    return paragraphs if paragraphs else [text] if text else []

def _semantic_search(self, paragraphs: list, question: str, top_k: int = 3) -> list:
    """
    语义相似度检索
    
    流程：
    1. 编码问题和段落为向量
    2. 计算余弦相似度
    3. 返回 Top-K 最相关的段落
    """
    if not SENTENCE_TRANSFORMERS_AVAILABLE or QAService._embedder is None:
        # 降级：返回前 N 段
        return paragraphs[:top_k]
    
    try:
        # 编码段落
        para_embeddings = QAService._embedder.encode(
            paragraphs, 
            convert_to_tensor=True
        )
        
        # 编码问题
        q_embedding = QAService._embedder.encode(
            question, 
            convert_to_tensor=True
        )
        
        # 语义搜索（计算余弦相似度并排序）
        hits = util.semantic_search(
            q_embedding, 
            para_embeddings, 
            top_k=top_k
        )[0]
        
        # 返回最相关的段落
        return [paragraphs[hit["corpus_id"]] for hit in hits]
        
    except Exception as e:
        logger.warning("语义检索失败，降级返回前 %d 段: %s", top_k, e)
        return paragraphs[:top_k]
```

```python
def _generate_answer(self, question: str, context: str) -> str:
    """
    使用 Qwen3 本地模型生成回答
    
    Prompt 设计：
    - 角色：会议纪要助手
    - 任务：根据会议文本回答问题
    - 约束：尽量从文本中找到相关信息
    """
    prompt = (
        f"你是会议纪要助手。根据以下会议文本回答问题。"
        f"即使信息不完全匹配，也请基于文本内容给出最相关的回答。\n\n"
        f"会议文本：\n{context}\n\n"
        f"问题：\n{question}\n\n"
        f"请直接回答："
    )
    
    try:
        # 导入本地 Qwen3 处理器
        from app.services.nlp_service import local_qwen_processor
        
        # 确保模型已加载
        local_qwen_processor._ensure_model()
        
        tokenizer = local_qwen_processor.__class__._tokenizer
        model = local_qwen_processor.__class__._model
        
        if tokenizer is None or model is None:
            return self._fallback_answer(question, context)
        
        # 使用 Qwen3 的 chat template
        messages = [{"role": "user", "content": prompt}]
        input_text = tokenizer.apply_chat_template(
            messages, 
            tokenize=False, 
            add_generation_prompt=True,
            enable_thinking=False  # 禁用思维链，加速推理
        )
        
        # 推理
        import torch
        inputs = tokenizer(
            input_text, 
            return_tensors="pt", 
            truncation=True, 
            max_length=8192
        )
        
        with torch.inference_mode():  # 推理模式，节省内存
            output = model.generate(
                **inputs, 
                max_new_tokens=512, 
                do_sample=False,  # 贪心解码
                pad_token_id=tokenizer.eos_token_id
            )
        
        # 解码输出
        response = tokenizer.decode(
            output[0][inputs["input_ids"].shape[1]:], 
            skip_special_tokens=True
        )
        
        return response.strip() if response.strip() else self._fallback_answer(question, context)
    
    except Exception as e:
        logger.warning("Qwen3 推理失败: %s", e)
        return self._fallback_answer(question, context)

def _fallback_answer(self, question: str, context: str) -> str:
    """
    兜底回答：基于关键词匹配
    
    当 Qwen3 不可用时的降级方案
    """
    # 提取问题中的关键词（长度 >= 2 的字符）
    keywords = [w for w in question if len(w) >= 2]
    
    # 在上下文中搜索包含关键词的句子
    matched = [line for line in context.split("\n") if any(k in line for k in keywords)]
    
    if matched:
        return "根据会议记录，相关段落如下：\n" + "\n".join(matched[:3])
    
    return "根据会议记录，与问题相关的信息如下：\n" + context[:500]
```

```python
            return {"success": False, "message": f"音频文件不存在: {input_path}"}

        if output_path is None:
            output_path = input_path

        try:
            # 1. 加载音频
            audio_segment = self._load_audio(input_path)
            if audio_segment is None:
                return {"success": False, "message": "无法加载音频文件"}

            # 2. 降噪
            if self.has_noisereduce:
                audio_segment = self._denoise(audio_segment)

            # 3. 音量标准化
            if self.has_pydub:
                audio_segment = self._normalize_volume(audio_segment)

            # 4. 回声消除
            if self.has_scipy:
                audio_segment = self._echo_cancellation(audio_segment)

            # 5. 导出
            audio_segment.export(output_path, format="wav")

            logger.info(f"音频增强完成: {output_path}")
            return {"success": True, "message": "音频增强完成", "output_path": output_path}

        except Exception as e:
            logger.error(f"音频增强失败: {e}")
            return {"success": False, "message": f"音频增强失败: {str(e)}"}
```

### 8.2 降噪处理

```python
def _denoise(self, audio_segment):
    """
    降噪处理 - 使用 noisereduce 库
    
    原理：基于频谱门控 (Spectral Gating)
    - 估计噪声的频谱特征
    - 在频域中抑制噪声分量
    - 逆变换回时域
    """
    try:
        import noisereduce as nr

        # 将 pydub 音频段转为 numpy 数组
        samples = np.array(audio_segment.get_array_of_samples(), dtype=np.float32)

        # 如果是立体声，转为单声道
        if audio_segment.channels == 2:
            samples = samples.reshape((-1, 2)).mean(axis=1)

        # 降噪
        reduced_samples = nr.reduce_noise(
            y=samples, 
            sr=audio_segment.frame_rate
        )

        # 转回 int16 格式
        reduced_samples = (reduced_samples * 32767).astype(np.int16)
        return audio_segment._spawn(reduced_samples.tobytes())

    except Exception as e:
        logger.warning(f"降噪失败，使用原始音频: {e}")
        return audio_segment
```

```python
def _normalize_volume(self, audio_segment, target_dbfs=-20):
    """
    音量标准化 - 使用 pydub 的 apply_gain
    
    原理：
    - 计算当前音频的平均音量 (dBFS)
    - 计算与目标音量的差值
    - 应用增益调整
    
    target_dbfs=-20 表示目标平均音量为 -20dBFS
    """
    try:
        # 计算当前音量
        current_dbfs = audio_segment.dBFS
        
        # 计算需要的增益
        change_in_dbfs = target_dbfs - current_dbfs
        
        # 应用增益
        return audio_segment.apply_gain(change_in_dbfs)
    except Exception as e:
        logger.warning(f"音量标准化失败: {e}")
        return audio_segment
```

### 8.4 回声消除（LMS 自适应滤波器）

```python
def _echo_cancellation(self, audio_segment):
    """
    回声消除 - 使用 LMS (Least Mean Squares) 自适应滤波器
    
    原理：
    - LMS 是一种自适应滤波算法
    - 通过迭代更新滤波器系数来逼近回声路径
    - 从原始信号中减去估计的回声
    
    参数说明：
    - delay_ms: 预估的回声延迟（毫秒）
    - filter_length: 滤波器长度（ taps 数量）
    - mu: 步长参数（学习率）
    
    算法流程：
    1. 构造参考信号（延迟版本，模拟回声）
    2. 初始化滤波器权重
    3. 迭代更新权重：
       - y = w^T * x     （滤波器输出，回声估计）
       - e = d - y        （误差信号，消除回声后的信号）
       - w = w + mu * e * x （更新权重）
    """
    try:
        import scipy.signal

        samples = np.array(audio_segment.get_array_of_samples(), dtype=np.float64)

        # 立体声转单声道
        if audio_segment.channels == 2:
            samples = samples.reshape((-1, 2)).mean(axis=1)

        sr = audio_segment.frame_rate
        delay_ms = 50                      # 预估回声延迟（毫秒）
        delay_samples = int(sr * delay_ms / 1000)
        filter_length = 128                 # 滤波器长度
        mu = 0.01                          # 步长参数（学习率）

        # 构造参考信号（延迟版本，模拟回声）
        ref = np.zeros(len(samples), dtype=np.float64)
        if delay_samples < len(samples):
            ref[delay_samples:] = samples[:-delay_samples]

        # LMS 自适应滤波
        weights = np.zeros(filter_length, dtype=np.float64)
        output = np.zeros(len(samples), dtype=np.float64)

        for i in range(filter_length, len(samples)):
            # 获取参考信号片段
            x = ref[i - filter_length:i][::-1].copy()
            
            # 滤波器输出（回声估计）
            y = np.dot(weights, x)
            
            # 误差信号（原始信号 - 回声估计 = 消除回声后的信号）
            e = samples[i] - y
            
            # 更新滤波器权重（LMS 算法）
            # w = w + mu * e * x
            weights += mu * e * x
            
            output[i] = e

        # 归一化并转回 int16
        max_val = np.max(np.abs(output)) if np.max(np.abs(output)) > 0 else 1.0
        output = (output / max_val * 32767).astype(np.int16)

        return audio_segment._spawn(output.tobytes())

    except Exception as e:
        logger.warning(f"回声消除失败，使用原始音频: {e}")
        return audio_segment
```

```python
class SpeakerDiarization:
    """
    基于声纹特征的说话人识别
    
    流程：
    1. 音频加载与预处理
    2. 语音活动检测 (VAD)
    3. 声纹特征提取（Resemblyzer）
    4. 聚类分析（层次聚类）
    """

    def __init__(self):
        self._check_dependencies()

    def _check_dependencies(self):
        """检查依赖库"""
        self.has_resemblyzer = False
        self.has_sklearn = False
        self.has_librosa = False

        try:
            from resemblyzer import VoiceEncoder
            self.has_resemblyzer = True
        except ImportError:
            logger.warning("resemblyzer 未安装，说话人识别不可用")

        try:
            from sklearn.cluster import AgglomerativeClustering
            self.has_sklearn = True
        except ImportError:
            logger.warning("scikit-learn 未安装，聚类分析不可用")

        try:
            import librosa
            self.has_librosa = True
        except ImportError:
            logger.warning("librosa 未安装，音频加载不可用")

    def diarize(self, audio_path: str) -> list:
        """
        对音频文件进行说话人识别
        
        Args:
            audio_path: 音频文件路径
        
        Returns:
            [{"speaker": "Speaker_1", "start": 0.0, "end": 2.5}, ...]
        """
        if not os.path.exists(audio_path):
            return [{"speaker": "未知发言人", "start": 0.0, "end": 0.0}]

        if not self.has_resemblyzer or not self.has_sklearn:
            return [{"speaker": "未知发言人", "start": 0.0, "end": 0.0}]

        try:
            # 1. 加载音频
            audio, duration = self._load_audio(audio_path)
            if audio is None:
                return [{"speaker": "未知发言人", "start": 0.0, "end": duration}]

            # 2. 语音活动检测
            segments = self._vad_segment(audio, duration)
            if not segments:
                return [{"speaker": "未知发言人", "start": 0.0, "end": duration}]

            # 3. 提取声纹嵌入
            embeddings = self._extract_embeddings(audio, segments)
            if len(embeddings) == 0:
                return [{"speaker": "未知发言人", "start": 0.0, "end": duration}]

            # 4. 聚类分析
            labels = self._cluster_speakers(embeddings)

            # 5. 组装结果
            result = []
            for i, (start, end) in enumerate(segments):
                if i < len(labels):
                    speaker = f"Speaker_{labels[i] + 1}"
                else:
                    speaker = "Speaker_1"
                result.append({
                    "speaker": speaker,
                    "start": round(start, 2),
                    "end": round(end, 2),
                })

            return result

        except Exception as e:
            logger.error(f"说话人识别失败: {e}")
            return [{"speaker": "未知发言人", "start": 0.0, "end": 0.0}]

    def _vad_segment(self, audio, duration: float) -> list:
        """
        语音活动检测 (VAD)
        
        基于能量阈值的端点检测：
        - 计算每帧的 RMS 能量
        - 超过阈值的帧标记为语音
        - 合并连续语音帧为段落
        """
        sample_rate = 16000
        frame_length = int(sample_rate * 0.025)  # 25ms 帧
        hop_length = int(sample_rate * 0.010)     # 10ms 跳跃
        
        segments = []
        in_speech = False
        start_time = 0.0
        
        for i in range(0, len(audio) - frame_length, hop_length):
            frame = audio[i:i + frame_length]
            
            # 计算 RMS 能量
            rms = np.sqrt(np.mean(frame ** 2))
            
            if rms > ENERGY_THRESHOLD:
                if not in_speech:
                    start_time = i / sample_rate
                    in_speech = True
            else:
                if in_speech:
                    end_time = i / sample_rate
                    if end_time - start_time >= MIN_SEGMENT_DURATION:
                        segments.append((start_time, end_time))
                    in_speech = False
        
        # 处理最后一段
        if in_speech:
            end_time = len(audio) / sample_rate
            if end_time - start_time >= MIN_SEGMENT_DURATION:
                segments.append((start_time, end_time))
        
        return segments

    def _extract_embeddings(self, audio, segments: list) -> list:
        """
        提取声纹嵌入向量
        
        使用 Resemblyzer 的 VoiceEncoder：
        - 基于 GE2E 损失训练的 LSTM
        - 输入：1.6 秒音频片段
        - 输出：256 维向量
        """
        from resemblyzer import VoiceEncoder

        encoder = VoiceEncoder()
        
        embeddings = []
        
        for start, end in segments:
            # 提取音频片段
            start_sample = int(start * SAMPLE_RATE)
            end_sample = int(end * SAMPLE_RATE)
            segment = audio[start_sample:end_sample]
            
            # Resemblyzer 要求 1.6 秒音频
            # 如果太短，填充静音；如果太长，截取中间部分
            target_length = int(1.6 * SAMPLE_RATE)
            
            if len(segment) < target_length:
                # 填充静音
                segment = np.pad(segment, (0, target_length - len(segment)))
            elif len(segment) > target_length:
                # 截取中间部分
                start_idx = (len(segment) - target_length) // 2
                segment = segment[start_idx:start_idx + target_length]
            
            # 编码为声纹向量
            embedding = encoder.embed_utterance(segment)
            embeddings.append(embedding)
        
        return embeddings

    def _cluster_speakers(self, embeddings: list) -> list:
        """
        使用层次聚类确定说话人
        
        层次聚类 (Agglomerative Clustering)：
        - 自底向上聚类
        - 使用平均链接 (Average Linkage)
        - 自动确定聚类数量
        """
        from sklearn.cluster import AgglomerativeClustering
        from sklearn.metrics.pairwise import cosine_similarity

        if len(embeddings) < 2:
            return [0] * len(embeddings)

        # 计算余弦相似度矩阵
        similarity_matrix = cosine_similarity(embeddings)
        
        # 转为距离矩阵（1 - 相似度）
        distance_matrix = 1 - similarity_matrix

        # 估算说话人数量
        n_speakers = self._estimate_speaker_count(distance_matrix)

        # 层次聚类
        clustering = AgglomerativeClustering(
            n_clusters=n_speakers,
            metric='precomputed',
            linkage='average'
        )

        labels = clustering.fit_predict(distance_matrix)
        return labels

    def _estimate_speaker_count(self, distance_matrix) -> int:
        """
        基于距离矩阵估算说话人数量
        
        启发式方法：
        - 计算平均距离
        - 如果平均距离较大，说明说话人差异大，可能有更多人
        """
        n = distance_matrix.shape[0]
        distances = []
        
        for i in range(n):
            for j in range(i + 1, n):
                distances.append(distance_matrix[i, j])
        
        if not distances:
            return 2
        
        avg_distance = np.mean(distances)
        
        # 简单启发式
        if avg_distance > 0.5:
            return min(5, n)  # 最多 5 人
        else:
            return max(2, n // 3)  # 至少 2 人
```

```python
class StreamingTranscriber:
    """
    基于 faster-whisper 的流式语音识别
    
    核心设计：
    - 缓冲区积累音频数据
    - 达到阈值后进行识别
    - 使用上下文提升连贯性
    """

    def __init__(self, sample_rate=16000, chunk_duration=3.0, overlap_duration=0.5):
        self.sample_rate = sample_rate
        self.chunk_duration = chunk_duration      # 每次识别的音频时长（秒）
        self.overlap_duration = overlap_duration  # 重叠时长（秒）
        self.buffer = bytearray()
        self._context_text = ""                   # 上下文文本
        self._input_sample_rate = None
        self._model = None                        # Whisper 模型

    def _init_model(self):
        """初始化 Whisper 模型"""
        if self._model is not None:
            return
        
        try:
            from faster_whisper import WhisperModel
            compute_type = "int8" if os.environ.get("LOCAL_MODEL_DEVICE", "cpu") == "cpu" else "float16"
            self._model = WhisperModel(
                WHISPER_MODEL_SIZE,
                device=os.environ.get("LOCAL_MODEL_DEVICE", "cpu"),
                compute_type=compute_type,
            )
            logger.info("Streaming whisper model loaded")
        except ImportError:
            logger.warning("faster-whisper not installed")

    def set_input_sample_rate(self, rate: int):
        """设置输入音频的采样率（浏览器通常是 44100 或 48000）"""
        self._input_sample_rate = rate

    def add_audio(self, audio_data: bytes):
        """添加音频数据到缓冲区"""
        self.buffer.extend(audio_data)

    def should_transcribe(self) -> bool:
        """检查缓冲区是否达到识别阈值"""
        input_rate = self._input_sample_rate or self.sample_rate
        expected_bytes = int(input_rate * 2 * (self.chunk_duration + self.overlap_duration))
        return len(self.buffer) >= expected_bytes

    def transcribe_chunk(self) -> dict:
        """
        识别当前缓冲区的音频
        
        流程：
        1. 将缓冲区转为 numpy 数组
        2. 重采样到 16kHz
        3. 静音检测
        4. 使用 faster-whisper 识别
        5. 更新上下文
        6. 清空缓冲区
        """
        self._init_model()
        
        if self._model is None:
            return {"text": "", "is_final": False}

        # 1. 将缓冲区转为 numpy 数组
        samples = np.frombuffer(self.buffer, dtype=np.int16)
        audio_np = samples.astype(np.float32) / 32768.0  # 归一化到 [-1, 1]

        # 2. 重采样到 16kHz
        audio_np = self._resample_to_16k(audio_np)

        # 3. 静音检测
        if self._is_silence(audio_np):
            self.buffer = bytearray()
            return {"text": "", "is_final": False}

        # 4. 使用 faster-whisper 识别
        segments, _ = self._model.transcribe(
            audio_np,
            language="zh",
            beam_size=5,
            initial_prompt=self._context_text  # 上下文提示
        )

        # 5. 拼接结果
        text = "".join([seg.text for seg in segments])

        # 6. 幻觉检测
        if self._is_hallucination(text):
            text = ""

        # 7. 更新上下文（保留最近 200 字符）
        self._context_text = (self._context_text + text)[-200:]

        # 8. 清空缓冲区
        self.buffer = bytearray()

        return {"text": text, "is_final": False}

    def _resample_to_16k(self, audio_np: np.ndarray) -> np.ndarray:
        """将音频重采样到 16kHz"""
        if self._input_sample_rate is None or self._input_sample_rate == 16000:
            return audio_np
        
        ratio = 16000 / self._input_sample_rate
        target_len = int(len(audio_np) * ratio)
        indices = np.linspace(0, len(audio_np) - 1, target_len).astype(int)
        return audio_np[indices]

    def _is_silence(self, audio_np: np.ndarray, threshold: float = 0.005) -> bool:
        """检测音频是否为静音（基于 RMS 能量）"""
        if len(audio_np) == 0:
            return True
        rms = np.sqrt(np.mean(audio_np ** 2))
        return rms < threshold

    def _is_hallucination(self, text: str) -> bool:
        """检测是否为 Whisper 幻觉输出"""
        if not text or not text.strip():
            return True
        t = text.strip()
        if len(t) < 2:
            return True
        if detect_repetition(t):
            return True
        if detect_prompt_echo(t):
            return True
        return False

    def reset(self):
        """重置状态"""
        self.buffer = bytearray()
        self._context_text = ""

    def get_final_text(self) -> str:
        """获取最终文本"""
        return self._context_text
```

```python
                f"问题：{request.question}\n\n"
                f"请直接回答："
            )

            # 使用 Qwen3 流式生成
            from app.services.nlp_service import local_qwen_processor
            local_qwen_processor._ensure_model()

            tokenizer = local_qwen_processor.__class__._tokenizer
            model = local_qwen_processor.__class__._model

            messages = [{"role": "user", "content": prompt}]
            input_text = tokenizer.apply_chat_template(
                messages, tokenize=False, add_generation_prompt=True, enable_thinking=False
            )

            import torch
            from transformers import TextIteratorStreamer
            import threading

            streamer = TextIteratorStreamer(tokenizer, skip_prompt=True, skip_special_tokens=True)

            inputs = tokenizer(input_text, return_tensors="pt", truncation=True, max_length=8192)

            # 在后台线程运行推理
            thread = threading.Thread(
                target=model.generate,
                kwargs={
                    "inputs": inputs,
                    "max_new_tokens": 512,
                    "streamer": streamer
                }
            )
            thread.start()

            # 逐块输出（SSE 格式）
            for text in streamer:
                yield f"data: {json.dumps({'chunk': text})}\n\n"

            yield f"data: {json.dumps({'done': True})}\n\n"

        except Exception as e:
            yield f"data: {json.dumps({'error': str(e)})}\n\n"

    return StreamingResponse(generate(), media_type="text/event-stream")
```

```javascript
                headers: { 'Content-Type': 'application/json' },

            this.currentMeetingId = data1.data.id;
            this.updateProgress('生成纪要…', 50, '正在提取关键信息…');

            // 生成会议纪要
            const resp2 = await fetch(API_BASE + `/api/meetings/${this.currentMeetingId}/generate-minutes`, {

            });
            const data2 = await resp2.json();

        try {
            const resp = await fetch(API_BASE + `/api/meetings/${this.currentMeetingId}/generate-minutes`, {
                method: 'POST',
            });
            const data = await resp.json();

            if (!data.success) throw new Error(data.message);

            this.completeProgress('纪要生成完成!');
            this.currentMinutes = data.data;
            this.actionItems = data.data.action_items || [];
            this._showMinutesResult(data.data);
            await this.loadHistory();

        } catch (err) {
            this.errorProgress(err.message);
            this.showToast(err.message, 'error');
```

```javascript
// 进度条相关方法

showProgress(label, percent, detail) {
    const el = this._els;
    el.progressSection.style.display = 'block';
    el.progressSection.classList.remove('completed', 'error');
    el.progressLabel.textContent = label;
    el.progressPercent.textContent = percent + '%';
    el.progressBar.style.width = percent + '%';
    el.progressDetail.textContent = detail;
}

updateProgress(label, percent, detail) {
    const el = this._els;
    el.progressLabel.textContent = label;
    el.progressPercent.textContent = percent + '%';
    el.progressBar.style.width = percent + '%';
    if (detail) el.progressDetail.textContent = detail;
}

completeProgress(detail) {
    const el = this._els;
    el.progressSection.classList.add('completed');
    el.progressBar.classList.remove('animating');
    this.updateProgress('完成', '100', detail);
}

errorProgress(detail) {
    const el = this._els;
    el.progressSection.classList.add('error');
    el.progressBar.classList.remove('animating');
    this.updateProgress('失败', '0', detail);
}

_startProgressAnimation() {
    this._els.progressBar.classList.add('animating');
    // 动态更新进度
    const stages = [
        { delay: 3000, pct: 20, detail: '正在处理音频数据...' },
        { delay: 8000, pct: 35, detail: '正在分析语音模式...' },
        { delay: 15000, pct: 50, detail: '正在转换为文字...' },
        { delay: 25000, pct: 65, detail: '正在优化转写结果...' },
        { delay: 40000, pct: 80, detail: '识别进行中，请稍候...' },
        { delay: 60000, pct: 90, detail: '即将完成...' },
    ];
    this._progressTimers = stages.map(s =>
        setTimeout(() => this.updateProgress('识别中', s.pct, s.detail), s.delay)
    );
}
```
