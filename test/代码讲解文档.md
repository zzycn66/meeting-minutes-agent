```python
@event.listens_for(engine, "connect")
def _set_sqlite_pragma(dbapi_connection, connection_record):
    cursor = dbapi_connection.cursor()
    cursor.execute("PRAGMA journal_mode=WAL")    # WAL 模式，支持并发读
    cursor.execute("PRAGMA synchronous=NORMAL")   # 平衡性能和安全
    cursor.execute("PRAGMA cache_size=-64000")    # 64MB 缓存
    cursor.execute("PRAGMA temp_store=MEMORY")    # 临时表存内存
```

```python
class SpeechRecognizer:
    """语音识别服务 - 双引擎架构"""
    
    def recognize_file(self, filepath: str, asr_engine: str = "") -> dict:
        """
        语音识别主流程
        
        Args:
            filepath: 音频文件路径
            asr_engine: 引擎选择 ("whisper" / "mimo" / "" 自动)
        """
        # 1. 音频预处理（可选增强）
        enhanced_path = self._enhance_if_needed(filepath)
        
        # 2. 根据引擎选择识别方法
        if asr_engine == "whisper" or not asr_engine:
            result = self._recognize_with_whisper(enhanced_path)
        elif asr_engine == "mimo":
            result = self._recognize_with_mimo(enhanced_path)
        
        # 3. 后处理
        cleaned_text = self._post_process(result["text"])
        
        return {"success": True, "text": cleaned_text}
```

```python
def _recognize_with_whisper(self, filepath: str) -> dict:
    """使用 faster-whisper 识别"""
    
    # 优先使用 faster-whisper（CTranslate2 加速，速度提升 2-4x）
    model = _get_faster_whisper_model()
    
    if model:
        # faster-whisper 识别
        segments, info = model.transcribe(
            filepath,
            language="zh",                    # 指定中文
            beam_size=5,                      # beam search 宽度
            best_of=5,                        # 取最佳 5 个结果
            temperature=0.0,                  # 贪心解码
            condition_on_previous_text=False,  # 不依赖前文（避免幻觉累积）
            initial_prompt=WHISPER_INITIAL_PROMPT,  # 初始提示词
        )
        
        # 拼接所有 segment 的文本
        text = "".join([seg.text for seg in segments])
    else:
        # 兜底：openai-whisper 原版
        model = _get_whisper_model()
        result = model.transcribe(filepath, **WHISPER_TRANSCRIBE_KWARGS)
        text = result["text"]
    
    return {"success": True, "text": text}
```

**讲解要点：**
- `faster-whisper` 基于 CTranslate2，使用 int8 量化，速度比原版快 2-4 倍
- `condition_on_previous_text=False` 防止幻觉在长文本中累积
- `initial_prompt` 提供上下文提示，提升中文识别准确率

### 3.4 幻觉检测与过滤

```python
def _filter_whisper_hallucinations(self, text: str) -> str:
    """过滤 Whisper 常见幻觉输出"""
    
    # 1. 检测 prompt echo（初始提示词回显）
    if detect_prompt_echo(text):
        return ""
    
    # 2. 检测重复文本（如 "哈哈哈哈哈哈哈"）
    if detect_repetition(text):
        text = self._remove_repetitions(text)
    
    # 3. 过滤无意义短句
    lines = text.split('\n')
    filtered = [line for line in lines if len(line.strip()) >= 3]
    
    return '\n'.join(filtered)

def detect_repetition(text: str) -> bool:
    """检测文本中的重复幻觉"""
    # 单字符重复 5 次以上
    if re.search(r'(.)\1{4,}', text):
        return True
    # 2-6 字符词重复 3 次以上
    if re.search(r'(\S{2,6})\1{3,}', text):
        return True
    # 短语重复 2 次以上
    if re.search(r'(.{2,10})\1{2,}', text):
        return True
    return False
```

### 4.2 规则匹配引擎

```python
def _rule_based_extract(self, text: str) -> dict:
    """基于规则的结构化信息提取"""
    
    # 关键决策关键词库
    DECISION_KEYWORDS = [
        "决定", "确定", "同意", "通过", "批准", "拍板", "敲定",
        "达成共识", "一致认为", "最终方案", "砍掉", "放弃",
        "推迟", "延后", "搁置", "优先做", "先做"
    ]
    
    # 行动项关键词库
    ACTION_KEYWORDS = [
        "负责", "由.*完成", "安排.*做", "指派", "落实", "执行",
        "推进", "下一步", "上线", "启动", "搞定", "出报告"
    ]
    
    # 提取关键决策
    decisions = []
    sentences = text.split('。')
    for sentence in sentences:
        if any(kw in sentence for kw in DECISION_KEYWORDS):
            decisions.append(sentence.strip())
    
    # 提取行动项
    action_items = []
    for sentence in sentences:
        if any(kw in sentence for kw in ACTION_KEYWORDS):
            item = self._parse_action_item(sentence)
            if item:
                action_items.append(item)
    
    return {
        "key_decisions": decisions,
        "action_items": action_items
    }

def _parse_action_item(self, sentence: str) -> dict:
    """解析行动项：提取责任人、截止时间"""
    
    # 人名提取模式
    PERSON_PATTERNS = [
        r'(?:由|请|让|安排|指派|交给)([\u4e00-\u9fa5]{2,4}?)(?:负责|完成|跟进)',
        r'([\u4e00-\u9fa5]{2,4})(?:负责|来负责|来完成)',
        r'@([\u4e00-\u9fa5]{2,4})',
    ]
    
    # 截止时间模式
    DEADLINE_PATTERNS = [
        r'(截止|deadline)[：:]\s*(.+?)(?:[。；\n]|$)',
        r'(\d+月\d+[日号])',
        r'(\d{4}[-/]\d{1,2}[-/]\d{1,2})',
        r'(下周[一二三四五六日])',
        r'(今天|明天|后天)',
    ]
    
    # 提取责任人
    responsible = ""
    for pattern in PERSON_PATTERNS:
        match = re.search(pattern, sentence)
        if match:
            responsible = match.group(1)
            break
    
    # 提取截止时间
    deadline = ""
    for pattern in DEADLINE_PATTERNS:
        match = re.search(pattern, sentence)
        if match:
            deadline = match.group(1) if match.lastindex else match.group(0)
            break
    
    return {
        "content": sentence,
        "responsible_person": responsible,
        "deadline": deadline
    }
```

```python
class MinutesGenerator:
    """会议纪要生成器"""
    
    TEMPLATE = """
# {title}

**会议时间**：{date}
**参会人员**：{participants}

---

## 一、会议议题
{topic}

## 二、会议概述
{overview}

## 三、讨论要点
{discussion_points}

## 四、关键决策
{key_decisions}

## 五、行动项
{action_items}

## 六、遗留问题
{unresolved_issues}

## 七、会议总结
{summary}

---
*纪要由会议纪要智能体自动生成*
*生成时间：{generated_at}*
"""
    
    def generate_markdown(self, meeting_data: dict) -> str:
        """生成 Markdown 格式纪要"""
        
        # 格式化行动项
        action_items = meeting_data.get("action_items", [])
        if action_items:
            action_lines = []
            for item in action_items:
                line = f"- [ ] {item['content']}"
                if item.get('responsible_person'):
                    line += f" —— **负责人**：{item['responsible_person']}"
                if item.get('deadline'):
                    line += f" **截止时间**：{item['deadline']}"
                if item.get('priority'):
                    line += f" [{item['priority']}]"
                action_lines.append(line)
            action_text = "\n".join(action_lines)
        else:
            action_text = "暂无行动项"
        
        # 格式化列表为文本
        def format_list(items):
            if isinstance(items, list):
                return "\n".join([f"- {item}" for item in items])
            return items or ""
        
        # 填充模板
        return self.TEMPLATE.format(
            title=meeting_data.get("title", "会议纪要"),
            date=meeting_data.get("meeting_date", ""),
            participants=meeting_data.get("participants", ""),
            topic=meeting_data.get("topic", ""),
            overview=meeting_data.get("overview", ""),
            discussion_points=format_list(meeting_data.get("discussion_points", "")),
            key_decisions=format_list(meeting_data.get("key_decisions", "")),
            action_items=action_text,
            unresolved_issues=format_list(meeting_data.get("unresolved_issues", "")),
            summary=meeting_data.get("summary", ""),
            generated_at=datetime.now().strftime("%Y-%m-%d %H:%M:%S")
```

### 6.2 语义检索

```python
def _split_paragraphs(self, text: str, min_len: int = 10) -> list:
    """将文本分割为段落"""
    paragraphs = []
    
    # 先按换行符分割
    for p in text.split("\n"):
        p = p.strip()
        if len(p) >= min_len:
            paragraphs.append(p)
    
    # 如果段落太少，按句子分割
    if not paragraphs:
        for sent in text.replace("。", "。\n").replace("！", "！\n").replace("？", "？\n").split("\n"):
            sent = sent.strip()
            if len(sent) >= min_len:
                paragraphs.append(sent)
    
    return paragraphs if paragraphs else [text] if text else []

def _semantic_search(self, paragraphs: list, question: str, top_k: int = 3) -> list:
    """语义相似度检索"""
    if not self._embedder:
        # 降级：返回前 N 段
        return paragraphs[:top_k]
    
    try:
        # 编码问题和段落为向量
        para_embeddings = self._embedder.encode(paragraphs, convert_to_tensor=True)
        q_embedding = self._embedder.encode(question, convert_to_tensor=True)
        
        # 计算余弦相似度并排序
        hits = util.semantic_search(q_embedding, para_embeddings, top_k=top_k)[0]
        
        # 返回最相关的段落
        return [paragraphs[hit["corpus_id"]] for hit in hits]
    except Exception as e:
        logger.warning("语义检索失败，降级返回前 %d 段: %s", top_k, e)
        return paragraphs[:top_k]
```

**讲解要点：**
- `paraphrase-multilingual-MiniLM-L12-v2` 是一个多语言句子嵌入模型，支持中文
- `semantic_search` 返回的是按相似度排序的 Top-K 结果
- 降级策略：语义检索失败时返回前 N 段

```python
def _generate_answer(self, question: str, context: str) -> str:
    """使用 Qwen3 本地模型生成回答"""
    
    prompt = (
        f"你是会议纪要助手。根据以下会议文本回答问题。"
        f"即使信息不完全匹配，也请基于文本内容给出最相关的回答。\n\n"
        f"会议文本：\n{context}\n\n"
        f"问题：\n{question}\n\n"
        f"请直接回答："
    )
    
    try:
        # 导入本地 Qwen3 处理器
        from app.services.nlp_service import local_qwen_processor
        
        # 确保模型已加载
        local_qwen_processor._ensure_model()
        
        tokenizer = local_qwen_processor.__class__._tokenizer
        model = local_qwen_processor.__class__._model
        
        if tokenizer is None or model is None:
            return self._fallback_answer(question, context)
        
        # 使用 Qwen3 的 chat template
        messages = [{"role": "user", "content": prompt}]
        input_text = tokenizer.apply_chat_template(
            messages, 
            tokenize=False, 
            add_generation_prompt=True,
            enable_thinking=False  # 禁用思维链，加速推理
        )
        
        # 推理
        import torch
        inputs = tokenizer(input_text, return_tensors="pt", truncation=True, max_length=8192)
        
        with torch.inference_mode():  # 推理模式，节省内存
            output = model.generate(
                **inputs, 
                max_new_tokens=512, 
                do_sample=False,  # 贪心解码
                pad_token_id=tokenizer.eos_token_id
            )
        
        # 解码输出
        response = tokenizer.decode(
            output[0][inputs["input_ids"].shape[1]:], 
            skip_special_tokens=True
        )
        
        return response.strip() if response.strip() else self._fallback_answer(question, context)
    
    except Exception as e:
        logger.warning("Qwen3 推理失败: %s", e)
        return self._fallback_answer(question, context)

def _fallback_answer(self, question: str, context: str) -> str:
    """兜底回答：基于关键词匹配"""
    keywords = [w for w in question if len(w) >= 2]
    matched = [line for line in context.split("\n") if any(k in line for k in keywords)]
    
    if matched:
        return "根据会议记录，相关段落如下：\n" + "\n".join(matched[:3])
    return "根据会议记录，与问题相关的信息如下：\n" + context[:500]
```

```python
def _denoise(self, audio_segment):
    """降噪处理 - 使用 noisereduce 库"""
    try:
        import noisereduce as nr
        
        # 将 pydub 音频段转为 numpy 数组
        samples = np.array(audio_segment.get_array_of_samples(), dtype=np.float32)
        
        # 如果是立体声，转为单声道
        if audio_segment.channels == 2:
            samples = samples.reshape((-1, 2)).mean(axis=1)
        
        # 降噪（基于频谱门控）
        reduced_samples = nr.reduce_noise(
            y=samples, 
            sr=audio_segment.frame_rate
        )
        
        # 转回 int16 格式
        reduced_samples = (reduced_samples * 32767).astype(np.int16)
        return audio_segment._spawn(reduced_samples.tobytes())
    
    except Exception as e:
        logger.warning(f"降噪失败，使用原始音频: {e}")
        return audio_segment
```

### 7.3 回声消除（LMS 自适应滤波器）

```python
def _echo_cancellation(self, audio_segment):
    """回声消除 - 使用 LMS 自适应滤波器"""
    try:
        import scipy.signal
        
        samples = np.array(audio_segment.get_array_of_samples(), dtype=np.float64)
        
        # 立体声转单声道
        if audio_segment.channels == 2:
            samples = samples.reshape((-1, 2)).mean(axis=1)
        
        sr = audio_segment.frame_rate
        delay_ms = 50           # 预估回声延迟（毫秒）
        delay_samples = int(sr * delay_ms / 1000)
        filter_length = 128     # 滤波器长度
        mu = 0.01               # 步长参数（学习率）
        
        # 构造参考信号（延迟版本，模拟回声）
        ref = np.zeros(len(samples), dtype=np.float64)
        if delay_samples < len(samples):
            ref[delay_samples:] = samples[:-delay_samples]
        
        # LMS 自适应滤波
        weights = np.zeros(filter_length, dtype=np.float64)
        output = np.zeros(len(samples), dtype=np.float64)
        
        for i in range(filter_length, len(samples)):
            # 获取参考信号片段
            x = ref[i - filter_length:i][::-1].copy()
            
            # 滤波器输出（回声估计）
            y = np.dot(weights, x)
            
            # 误差信号（原始信号 - 回声估计 = 消除回声后的信号）
            e = samples[i] - y
            
            # 更新滤波器权重（LMS 算法）
            weights += mu * e * x
            
            output[i] = e
        
        # 归一化并转回 int16
        max_val = np.max(np.abs(output)) if np.max(np.abs(output)) > 0 else 1.0
        output = (output / max_val * 32767).astype(np.int16)
        
        return audio_segment._spawn(output.tobytes())
    
    except Exception as e:
        logger.warning(f"回声消除失败，使用原始音频: {e}")
        return audio_segment
```

**讲解要点：**
- **LMS (Least Mean Squares)** 是一种自适应滤波算法
- `ref` 是延迟版本的原始信号，模拟回声
- `weights` 是滤波器系数，通过迭代更新来逼近回声路径
- `e = samples[i] - y` 是误差信号，即消除回声后的输出
- `mu` 控制收敛速度和稳定性，太大会发散，太小收敛慢

### 8.2 层次聚类

```python
def _cluster_speakers(self, embeddings, n_speakers=None):
    """使用层次聚类确定说话人"""
    from sklearn.cluster import AgglomerativeClustering
    
    if len(embeddings) < 2:
        return [0] * len(embeddings)
    
    # 计算余弦相似度矩阵
    from sklearn.metrics.pairwise import cosine_similarity
    similarity_matrix = cosine_similarity(embeddings)
    
    # 转为距离矩阵
    distance_matrix = 1 - similarity_matrix
    
    # 如果未指定说话人数量，使用距离阈值自动确定
    if n_speakers is None:
        n_speakers = self._estimate_speaker_count(distance_matrix)
    
    # 层次聚类
    clustering = AgglomerativeClustering(
        n_clusters=n_speakers,
        metric='precomputed',
        linkage='average'
    )
    
    labels = clustering.fit_predict(distance_matrix)
    return labels

def _estimate_speaker_count(self, distance_matrix):
    """基于距离矩阵估算说话人数量"""
    # 提取上三角距离
    n = distance_matrix.shape[0]
    distances = []
    for i in range(n):
        for j in range(i+1, n):
            distances.append(distance_matrix[i, j])
    
    # 使用阈值判断
    threshold = 0.5  # 可调整
    avg_distance = np.mean(distances)
    
    # 简单启发式：距离差异大说明说话人多
    if avg_distance > threshold:
        return min(5, n)  # 最多 5 人
    else:
        return max(2, n // 3)  # 至少 2 人
```

```python
class StreamingTranscriber:
    """基于 faster-whisper 的流式语音识别"""
    
    def __init__(self, sample_rate=16000, chunk_duration=3.0):
        self.sample_rate = sample_rate
        self.chunk_duration = chunk_duration  # 每次识别的音频时长（秒）
        self.buffer = bytearray()
        self._context_text = ""  # 上下文文本，提升识别连贯性
    
    def add_audio(self, audio_data: bytes):
        """添加音频数据到缓冲区"""
        self.buffer.extend(audio_data)
    
    def should_transcribe(self) -> bool:
        """检查缓冲区是否达到识别阈值"""
        # 16-bit mono，每秒 32000 字节
        expected_bytes = int(self.sample_rate * 2 * self.chunk_duration)
        return len(self.buffer) >= expected_bytes
    
    def transcribe_chunk(self) -> dict:
        """识别当前缓冲区的音频"""
        # 1. 将缓冲区转为 numpy 数组
        audio_np = self._buffer_to_numpy()
        
        # 2. 重采样到 16kHz（Whisper 要求）
        audio_np = self._resample_to_16k(audio_np)
        
        # 3. 静音检测
        if self._is_silence(audio_np):
            self.buffer = bytearray()
            return {"text": "", "is_final": False}
        
        # 4. 使用 faster-whisper 识别
        segments, _ = self._model.transcribe(
            audio_np,
            language="zh",
            beam_size=5,
            initial_prompt=self._context_text  # 上下文提示
        )
        
        # 5. 拼接结果
        text = "".join([seg.text for seg in segments])
        
        # 6. 幻觉检测
        if self._is_hallucination(text):
            text = ""
        
        # 7. 更新上下文（保留最近 200 字符）
        self._context_text = (self._context_text + text)[-200:]
        
        # 8. 清空缓冲区
        self.buffer = bytearray()
        
        return {"text": text, "is_final": False}
    
    def _buffer_to_numpy(self):
        """将字节缓冲区转为 numpy 数组"""
        # 16-bit PCM，小端序
        samples = np.frombuffer(self.buffer, dtype=np.int16)
        # 归一化到 [-1, 1]
        return samples.astype(np.float32) / 32768.0
    
    def _resample_to_16k(self, audio_np):
        """重采样到 16kHz"""
        if self._input_sample_rate == 16000:
            return audio_np
        
        ratio = 16000 / self._input_sample_rate
        target_len = int(len(audio_np) * ratio)
        indices = np.linspace(0, len(audio_np) - 1, target_len).astype(int)
        return audio_np[indices]
```

```python
        # 逐块输出
        for text in streamer:
            yield f"data: {json.dumps({'chunk': text})}\n\n"
        
        yield f"data: {json.dumps({'done': True})}\n\n"
    
    return StreamingResponse(generate(), media_type="text/event-stream")
```

## 十一、WebSocket 实时转写 (`websocket.py`)

### 11.1 WebSocket 路由

```python
# routes/websocket.py

@router.websocket("/ws/transcribe")
async def websocket_transcribe(websocket: WebSocket):
    """WebSocket 实时转写"""
    
    # 认证
    if not await _verify_ws_auth(websocket):
        return
    
    await websocket.accept()
    transcriber = StreamingTranscriber()
    
    try:
        while True:
            data = await websocket.receive()
            
            if data["type"] == "websocket.receive":
                if "text" in data:
                    msg = json.loads(data["text"])
                    
                    if msg.get("type") == "start":
                        # 开始转写
                        transcriber.reset()
                        if msg.get("sample_rate"):
                            transcriber.set_input_sample_rate(msg["sample_rate"])
                    
                    elif msg.get("type") == "stop":
                        # 停止转写，发送最终结果
                        final_text = transcriber.get_final_text()
                        await websocket.send_json({
                            "type": "transcript",
                            "text": final_text,
                            "is_final": True
                        })
                
                elif "bytes" in data:
                    # 接收音频数据
                    audio_data = data["bytes"]
                    transcriber.add_audio(audio_data)
                    
                    # 检查是否达到识别阈值
                    if transcriber.should_transcribe():
                        result = transcriber.transcribe_chunk()
                        
                        # 发送识别结果
                        await websocket.send_json({
                            "type": "transcript",
                            "text": result["text"],
                            "is_final": False
                        })
    
    except WebSocketDisconnect:
        logger.info("WebSocket disconnected")
```
